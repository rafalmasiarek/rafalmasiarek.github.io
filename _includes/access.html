Recommended for managed servers. Install the role directly from Git, then call it in your playbook. Replace rm only if you need a different account name.
requirements.yml 
roles:
  - name: rm_access
    src: git+https://github.com/rafalmasiarek/rm-ansible-access-key-role.git
    version: main
Install the role 
ansible-galaxy role install -r requirements.yml
Playbook: add access 
- name: Install SSH access
  hosts: all
  become: true

  roles:
    - role: rm_access
      vars:
        rm_access_action: add
        rm_access_username: rm
        rm_access_sudo: true
Playbook: remove access 
- name: Remove SSH access
  hosts: all
  become: true

  roles:
    - role: rm_access
      vars:
        rm_access_action: remove
        rm_access_username: rm
Playbook: update key only 
- name: Update SSH key only
  hosts: all
  become: false

  roles:
    - role: rm_access
      vars:
        rm_access_action: update
Recommended for new EC2 instances. Use user_data with cloud-init runcmd so access is configured during first boot.
aws_instance + cloud-init (add) 
resource "aws_instance" "rm_access" {
  ami                    = "ami-xxxxxxxxxxxxxxxxx"
  instance_type          = "t3.micro"
  subnet_id              = "subnet-xxxxxxxx"
  vpc_security_group_ids = ["sg-xxxxxxxx"]

  user_data = <<-EOF
    #cloud-config
    runcmd:
      - |
        set -euo pipefail
        t=$(mktemp)
        e=$(curl -fsSL -D - https://access.masiarek.pl -o "$t" | tr -d '\r' | sed -n 's/^X-Script-Fingerprint: sha256://Ip' | tail -n1)
        test -n "$e"
        test "$(sha256sum "$t" | cut -d' ' -f1)" = "$e"
        bash "$t" -- add --username rm --sudo
        rm -f "$t"
  EOF

  tags = {
    Name = "rm-access-ec2"
  }
}

Salt can run the installer through cmd.run. These examples verify the published script hash before execution.
State: add 
ssh_access_add:
  cmd.run:
    - name: |
        set -euo pipefail
        t=$(mktemp)
        e=$(curl -fsSL -D - https://access.masiarek.pl -o "$t" | tr -d '\r' | sed -n 's/^X-Script-Fingerprint: sha256://Ip' | tail -n1)
        test -n "$e"
        test "$(sha256sum "$t" | cut -d' ' -f1)" = "$e"
        sudo bash "$t" -- add --username rm --sudo
        rm -f "$t"
    - shell: /bin/bash
